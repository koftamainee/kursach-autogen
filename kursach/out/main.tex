\begin{algorithm}
\caption{Двоичный поиск}\label{alg.search}
\begin{algorithmic}[1]
  \Require отсортированный массив $A$, целевое значение $x$
  \State $lo \gets 0$, $hi \gets n-1$
  \While{$lo \leq hi$}
    \State $mid \gets \lfloor(lo+hi)/2\rfloor$
    \If{$A[mid] = x$}
      \State \Return $mid$
    \Else
      \If{$A[mid] < x$}
        \State $lo \gets mid + 1$
      \Else
        \State $hi \gets mid - 1$
      \EndIf
    \EndIf
  \EndWhile
  \State \Return $-1$
  \Comment{Поиск завершён}
  \Ensure индекс $x$ или $-1$
  \For{$i \in 1..n$}
    \State печать $i$
  \EndFor
\end{algorithmic}
\end{algorithm}
